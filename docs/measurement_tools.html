 26    def __init__(self, plot: pg.PlotItem, on_line_drawn_callback, on_drawing_state_changed=None):
 27        """Initialize line-drawing tool state and event handler references.
 28
 29        Args:
 30            plot: Plot item where measurement lines are drawn.
 31            on_line_drawn_callback: Callback invoked with (start_point, end_point).
 32            on_drawing_state_changed: Optional callback receiving drawing-active state.
 33        """
 34        self.plot = plot
 35        self.on_line_drawn_callback = on_line_drawn_callback
 36        self.on_drawing_state_changed = on_drawing_state_changed
 37        self.drawing = False
 38        self._last_drawing_state = False
 39        self.start_point: Optional[Tuple[float, float]] = None
 40        self.line_item: Optional[pg.PlotDataItem] = None
 41        self.is_enabled = False
 42        self.vb = plot.vb
 43        self._previous_mouse_enabled: Optional[Tuple[bool, bool]] = None
 44
 45        # Store original mouse event methods
 46        self.original_mouse_press = self.vb.mousePressEvent
 47        self.original_mouse_move = self.vb.mouseMoveEvent
 48        self.original_mouse_release = self.vb.mouseReleaseEvent
 49        self.original_mouse_drag = getattr(self.vb, "mouseDragEvent", None)
 50
 51    def _set_fft_roi_interaction_toggle(self, enabled: bool) -> None:
 52        """Enable/disable mouse interaction on FFT ROI items while measuring.
 53
 54                        Args:
 55                            enabled: Input value for enabled.
 56                    
 57        """
 58        items = getattr(self.plot, "items", [])
 59        for item in items:
 60            if isinstance(item, FFTBoxROI):
 61                try:
 62                    item.set_interaction_enabled(enabled)
 63                except Exception:
 64                    pass

 66    def enable(self):
 67        """Enable line drawing mode."""
 68        self.is_enabled = True
 69        self._set_drawing_state(False)
 70        self.start_point = None
 71        logger.debug("LineDrawingTool enabled")
 72
 73        # Disable ViewBox panning/zoom interaction while measuring.
 74        try:
 75            enabled = getattr(self.vb, "state", {}).get("mouseEnabled", [True, True])
 76            if isinstance(enabled, (list, tuple)) and len(enabled) >= 2:
 77                self._previous_mouse_enabled = (bool(enabled[0]), bool(enabled[1]))
 78            else:
 79                self._previous_mouse_enabled = (True, True)
 80            self.vb.setMouseEnabled(x=False, y=False)
 81        except Exception:
 82            self._previous_mouse_enabled = None
 83
 84        # Disable ROI interaction so ROI items don't steal mouse events.
 85        self._set_fft_roi_interaction_toggle(False)
 86
 87        # Replace mouse event handlers
 88        self.vb.mousePressEvent = self._on_mouse_press
 89        self.vb.mouseMoveEvent = self._on_mouse_move
 90        self.vb.mouseReleaseEvent = self._on_mouse_release
 91        if self.original_mouse_drag is not None:
 92            self.vb.mouseDragEvent = self._on_mouse_drag
 93
 94    def disable(self):
 95        """Disable line drawing mode."""
 96        self.is_enabled = False
 97
 98        # Restore original mouse event handlers
 99        self.vb.mousePressEvent = self.original_mouse_press
100        self.vb.mouseMoveEvent = self.original_mouse_move
101        self.vb.mouseReleaseEvent = self.original_mouse_release
102        if self.original_mouse_drag is not None:
103            self.vb.mouseDragEvent = self.original_mouse_drag
104
105        # Restore ViewBox panning/zoom interaction when leaving measurement mode.
106        if self._previous_mouse_enabled is not None:
107            try:
108                self.vb.setMouseEnabled(
109                    x=self._previous_mouse_enabled[0],
110                    y=self._previous_mouse_enabled[1],
111                )
112            except Exception:
113                pass
114            self._previous_mouse_enabled = None
115
116        # Re-enable ROI interaction when not measuring.
117        self._set_fft_roi_interaction_toggle(True)
118
119        self._clear_preview_line()
120        self._set_drawing_state(False)
121        self.start_point = None
122        logger.debug("LineDrawingTool disabled")
123
124    def _set_drawing_state(self, is_drawing: bool) -> None:
125        """Update drawing-state flag and notify listeners when it changes.
126
127                        Args:
128                            is_drawing: Boolean flag indicating whether drawing.
129                    
130        """
131        self.drawing = is_drawing
132        if self._last_drawing_state == is_drawing:
133            return
134        self._last_drawing_state = is_drawing
135        logger.debug("LineDrawingTool drawing state changed: drawing=%s", is_drawing)
136        if self.on_drawing_state_changed is not None:
137            self.on_drawing_state_changed(is_drawing)

139    def _clear_preview_line(self):
140        """Remove preview line from plot."""
141        if self.line_item is not None:
142            self.plot.removeItem(self.line_item)
143            self.line_item = None

145    def _begin_line(self, view_pos) -> None:
146        """Start a new preview line from the given view position.
147
148                        Args:
149                            view_pos: Input value for view pos.
150                    
151        """
152        self._set_drawing_state(True)
153        self.start_point = (view_pos.x(), view_pos.y())
154        logger.debug("LineDrawingTool start point: %s", self.start_point)
155        self._clear_preview_line()
156
157    def _update_line_preview(self, view_pos) -> None:
158        """Redraw the temporary preview segment to the current cursor point.
159
160                        Args:
161                            view_pos: Input value for view pos.
162                    
163        """
164        if self.start_point is None:
165            return
166        self._clear_preview_line()
167        self.line_item = pg.PlotDataItem(
168            [self.start_point[0], view_pos.x()],
169            [self.start_point[1], view_pos.y()],
170            pen=PREVIEW_LINE_PEN,
171        )
172        self.plot.addItem(self.line_item)
173
174    def _finish_line(self, view_pos) -> None:
175        """Finalize the current line and trigger the line-drawn callback.
176
177                        Args:
178                            view_pos: Input value for view pos.
179                    
180        """
181        if self.start_point is None:
182            return
183        end_point = (view_pos.x(), view_pos.y())
184        logger.debug("LineDrawingTool end point: %s", end_point)
185        self._set_drawing_state(False)
186        self._clear_preview_line()
187        self.on_line_drawn_callback(self.start_point, end_point)
188        self.start_point = None
189
190    def _on_mouse_press(self, event):
191        """Handle mouse press for line drawing.
192
193                        Args:
194                            event: Qt event object carrying user interaction details.
195                    
196        """
197        if not self.is_enabled:
198            self.original_mouse_press(event)
199            return
200
201        scene_pos = event.scenePos()
202        if not self.plot.sceneBoundingRect().contains(scene_pos):
203            self.original_mouse_press(event)
204            return
205
206        view_pos = self.vb.mapSceneToView(scene_pos)
207        # Start a new line on mouse press
208        self._begin_line(view_pos)
209
210        event.accept()
211
212    def _on_mouse_move(self, event):
213        """Handle mouse move for line drawing preview.
214
215                        Args:
216                            event: Qt event object carrying user interaction details.
217                    
218        """
219        if not self.is_enabled:
220            self.original_mouse_move(event)
221            return
222
223        if not self.drawing or self.start_point is None:
224            self.original_mouse_move(event)
225            return
226
227        scene_pos = event.scenePos()
228        if not self.plot.sceneBoundingRect().contains(scene_pos):
229            self.original_mouse_move(event)
230            return
231
232        view_pos = self.vb.mapSceneToView(scene_pos)
233        self._update_line_preview(view_pos)
234        event.accept()
235
236    def _on_mouse_release(self, event):
237        """Handle mouse release.
238
239                        Args:
240                            event: Qt event object carrying user interaction details.
241                    
242        """
243        if not self.is_enabled:
244            self.original_mouse_release(event)
245            return
246
247        # If we weren't drawing, let the original handler process the event
248        if not self.drawing or self.start_point is None:
249            self.original_mouse_release(event)
250            return
251
252        scene_pos = event.scenePos()
253        if not self.plot.sceneBoundingRect().contains(scene_pos):
254            # Outside plot area; cancel the drawing and delegate
255            self._clear_preview_line()
256            self._set_drawing_state(False)
257            self.start_point = None
258            logger.debug("LineDrawingTool draw cancelled: mouse release outside plot area")
259            self.original_mouse_release(event)
260            return
261
262        # Finish the line on mouse release and trigger callback
263        view_pos = self.vb.mapSceneToView(scene_pos)
264        self._finish_line(view_pos)
265
266        event.accept()
267
268    def _on_mouse_drag(self, event):
269        """Handle drag events directly so measurement works after prior pan/ROI drags.
270
271                        Args:
272                            event: Qt event object carrying user interaction details.
273                    
274        """
275        if not self.is_enabled:
276            if self.original_mouse_drag is not None:
277                self.original_mouse_drag(event)
278            return
279
280        scene_pos = event.scenePos()
281        if not self.plot.sceneBoundingRect().contains(scene_pos):
282            if event.isFinish() and self.drawing:
283                self._clear_preview_line()
284                self._set_drawing_state(False)
285                self.start_point = None
286                logger.debug("LineDrawingTool draw cancelled: drag finished outside plot area")
287            event.accept()
288            return
289
290        view_pos = self.vb.mapSceneToView(scene_pos)
291        if event.isStart() or self.start_point is None:
292            self._begin_line(view_pos)
293        elif event.isFinish():
294            self._finish_line(view_pos)
295        else:
296            self._update_line_preview(view_pos)
297
298        event.accept()

307    def set_interaction_enabled(self, enabled: bool) -> None:
308        """Toggle interaction for ROI and its resize/rotate handle items.
309
310                        Args:
311                            enabled: Input value for enabled.
312                    
313        """
314        mouse_buttons = QtCore.Qt.LeftButton if enabled else QtCore.Qt.NoButton
315
316        try:
317            self.setAcceptedMouseButtons(mouse_buttons)
318        except Exception:
319            pass
320
321        try:
322            self.setAcceptHoverEvents(enabled)
323        except Exception:
324            pass
325
326        if not enabled:
327            try:
328                self.setSelected(False)
329            except Exception:
330                pass
331
332        for child in list(self.childItems()):
333            try:
334                child.setAcceptedMouseButtons(mouse_buttons)
335            except Exception:
336                pass
337            try:
338                child.setAcceptHoverEvents(enabled)
339            except Exception:
340                pass

342    def mouseClickEvent(self, ev):  # type: ignore[override]
343        """Emit signals on single and double clicks while preserving default behavior.
344
345                        Args:
346                            ev: Input value for ev.
347                    
348        """
349        # Call base implementation if available (version-safe)
350        try:
351            super().mouseClickEvent(ev)  # type: ignore[misc]
352        except AttributeError:
353            pass
354
355        if ev.button() == QtCore.Qt.LeftButton:
356            if ev.double():
357                self.sigBoxDoubleClicked.emit(self)
358            else:
359                self.sigBoxClicked.emit(self)
