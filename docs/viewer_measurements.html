  95                self.logger.debug("Restoring calibration dialog with pending state")
  96                QtCore.QTimer.singleShot(0, lambda: viewer._open_calibration_dialog(state))

  98        if viewer.btn_measure is not None and viewer.btn_measure.isChecked():
  99            viewer.btn_measure.setChecked(False)
 100
 101        if viewer.line_tool is not None:
 102            viewer.line_tool.disable()
 103        viewer._on_measurement_drawing_state_changed(False)
 104        if viewer.btn_measure is not None:
 105            viewer.btn_measure.setStyleSheet("")
 106        self.logger.debug("Measurement mode exited")

 174    def on_line_drawn(self, p1: Tuple[float, float], p2: Tuple[float, float]) -> None:
 175        """Handle a line being drawn for measurement.
 176        
 177        Processes the line based on the current drawing mode:
 178        - "profile": Adds a line profile measurement.
 179        - "calibration": Converts physical distance to pixel distance via callback.
 180        - "measurement": Adds a distance or FFT frequency measurement.
 181        
 182        Args:
 183            p1: Start point of the line (x, y).
 184            p2: End point of the line (x, y).
 185        """
 186        viewer = self.viewer
 187        self.logger.debug("Line drawn: mode=%s p1=%s p2=%s", viewer._line_draw_mode, p1, p2)
 188        if viewer._line_draw_mode == "profile":
 189            self._add_profile_measurement(p1, p2)
 190            return
 191
 192        if viewer._line_draw_mode == "calibration":
 193            scale_x = float(viewer.ax_x.scale) if viewer.ax_x is not None else 1.0
 194            scale_y = float(viewer.ax_y.scale) if viewer.ax_y is not None else scale_x
 195
 196            dx_phys = float(p2[0] - p1[0])
 197            dy_phys = float(p2[1] - p1[1])
 198            if scale_x != 0 and scale_y != 0:
 199                dx_px = dx_phys / scale_x
 200                dy_px = dy_phys / scale_y
 201                dist_px = float(np.hypot(dx_px, dy_px))
 202            else:
 203                dist_px = 0.0
 204
 205            viewer._line_draw_mode = "measurement"
 206            if viewer.line_tool is not None:
 207                viewer.line_tool.disable()
 208
 209            callback = viewer._on_calibration_pixels_selected
 210            viewer._on_calibration_pixels_selected = None
 211            if callback is not None:
 212                self.logger.debug("Calibration distance selected: %.6g px", dist_px)
 213                callback(dist_px)
 214            return
 215
 216        viewer.measurement_count += 1
 217        measurement_id = viewer.measurement_count
 218
 219        if viewer.view_mode == "fft":
 220            dx_freq = float(p2[0] - p1[0])
 221            dy_freq = float(p2[1] - p1[1])
 222            dist_freq = float(np.hypot(dx_freq, dy_freq))
 223
 224            if viewer._nyq_x and viewer._nyq_y and viewer._fft_region is not None:
 225                px_scale_x = (2.0 * float(viewer._nyq_x)) / float(viewer._fft_region.shape[1])
 226                px_scale_y = (2.0 * float(viewer._nyq_y)) / float(viewer._fft_region.shape[0])
 227                if px_scale_x != 0 and px_scale_y != 0:
 228                    dx_px = dx_freq / px_scale_x
 229                    dy_px = dy_freq / px_scale_y
 230                    dist_px = float(np.hypot(dx_px, dy_px))
 231                else:
 232                    dist_px = 0.0
 233            else:
 234                dist_px = 0.0
 235
 236            result: Dict[str, float] = {
 237                "distance_pixels": dist_px,
 238                "distance_physical": dist_freq,
 239                "scale_x": float(viewer.ax_x.scale) if viewer.ax_x else 1.0,
 240                "scale_y": float(viewer.ax_y.scale) if viewer.ax_y else 1.0,
 241            }
 242            if dist_freq != 0:
 243                result["d_spacing"] = utils.calculate_d_spacing(dist_freq)
 244
 245            self._add_measurement_graphics(p1, p2, result, measurement_id)
 246            self.logger.debug(
 247                "FFT measurement #%s stored: distance_physical=%.6g distance_pixels=%.6g",
 248                measurement_id,
 249                dist_freq,
 250                dist_px,
 251            )
 252            return
 253
 254        scale_x = float(viewer.ax_x.scale) if viewer.ax_x is not None else 1.0
 255        scale_y = float(viewer.ax_y.scale) if viewer.ax_y is not None else scale_x
 256
 257        dx_phys = float(p2[0] - p1[0])
 258        dy_phys = float(p2[1] - p1[1])
 259        dist_phys = float(np.hypot(dx_phys, dy_phys))
 260
 261        if scale_x != 0 and scale_y != 0:
 262            dx_px = dx_phys / scale_x
 263            dy_px = dy_phys / scale_y
 264            dist_px = float(np.hypot(dx_px, dy_px))
 265        else:
 266            dist_px = 0.0
 267
 268        result = {
 269            "distance_pixels": dist_px,
 270            "distance_physical": dist_phys,
 271            "scale_x": scale_x,
 272            "scale_y": scale_y,
 273        }
 274
 275        if viewer.is_reciprocal_space and dist_phys != 0:
 276            frequency = 1.0 / dist_phys
 277            result["d_spacing"] = utils.calculate_d_spacing(frequency)
 278
 279        self._add_measurement_graphics(p1, p2, result, measurement_id)
 280        self.logger.debug(
 281            "Measurement #%s: p1=%s p2=%s distance_physical=%.6g distance_pixels=%.6g",
 282            measurement_id,
 283            p1,
 284            p2,
 285            dist_phys,
 286            dist_px,
 287        )

 339    def delete_selected_measurement(self) -> None:
 340        """Delete the currently selected measurement from the viewer.
 341        
 342        Removes the selected measurement's line and label from the plot.
 343        Does nothing if no measurement is selected or the index is invalid.
 344        """
 345        viewer = self.viewer
 346        if viewer.selected_measurement_index is None:
 347            self.logger.debug("Delete selected measurement ignored: nothing selected")
 348            return
 349
 350        if not (0 <= viewer.selected_measurement_index < len(viewer.measurement_items)):
 351            self.logger.debug(
 352                "Delete selected measurement ignored: invalid index=%s count=%s",
 353                viewer.selected_measurement_index,
 354                len(viewer.measurement_items),
 355            )
 356            viewer.selected_measurement_index = None
 357            return
 358
 359        deleted_index = viewer.selected_measurement_index
 360        measurement_id, line_item, text_item = viewer.measurement_items.pop(viewer.selected_measurement_index)
 361        viewer.p1.removeItem(line_item)
 362        viewer.p1.removeItem(text_item)
 363
 364        viewer.selected_measurement_index = None
 365        for _measurement_id, _line_item, item in viewer.measurement_items:
 366            self.set_label_fill(item, LABEL_BRUSH_COLOR)
 367        self.logger.debug(
 368            "Deleted selected measurement: index=%s id=%s remaining=%s",
 369            deleted_index,
 370            measurement_id,
 371            len(viewer.measurement_items),
 372        )

 387    def delete_measurement_by_label(self, label_text: str) -> None:
 388        """Delete a measurement by its label text.
 389        
 390        Args:
 391            label_text: The label text to delete (e.g., "M#5" or "P#3").
 392        """
 393        viewer = self.viewer
 394        self.logger.debug("Delete measurement by label requested: %s", label_text)
 395        target_index = None
 396        for idx, (_measurement_id, _line_item, text_item) in enumerate(viewer.measurement_items):
 397            if text_item.toPlainText() == label_text:
 398                target_index = idx
 399                break
 400
 401        if target_index is not None:
 402            viewer.selected_measurement_index = target_index
 403            self.delete_selected_measurement()
 404            return
 405
 406        if label_text.startswith("P#"):
 407            profile_id_text = label_text.split(" ", 1)[0][2:]
 408            try:
 409                profile_id = int(profile_id_text)
 410            except Exception:
 411                profile_id = None
 412
 413            if profile_id is not None:
 414                deleted = self._delete_profile_measurement(profile_id)
 415                if deleted:
 416                    self.logger.debug("Deleted profile by label: id=%s", profile_id)
 417                    return
 418
 419        self.logger.debug("Delete measurement by label ignored: not found")

 452    def open_measurement_by_history_id(self, entry_id: int, entry_type: str) -> None:
 453        """Open/display a measurement from the history window.
 454        
 455        Selects the measurement to display its profile plot (for profiles) or
 456        highlights the measurement on the main image (for distances).
 457        
 458        Args:
 459            entry_id: The ID of the measurement from the history window.
 460            entry_type: Type of measurement ("distance" or "profile").
 461        """
 462        viewer = self.viewer
 463        self.logger.debug(
 464            "Open measurement by history id requested: id=%s type=%s",
 465            entry_id,
 466            entry_type,
 467        )
 468        if entry_type != "profile":
 469            self.logger.debug("Open measurement ignored: non-profile entry")
 470            return
 471
 472        record = viewer.profile_measurement_items.get(entry_id)
 473        if record is None:
 474            self.logger.debug("Open profile ignored: id not found")
 475            return
 476
 477        window = record.get("window")
 478        if isinstance(window, LineProfileWindow):
 479            try:
 480                if window.isVisible():
 481                    window.raise_()
 482                    window.activateWindow()
 483                    self.logger.debug("Focused open profile window: id=%s", entry_id)
 484                    return
 485                window.show()
 486                window.raise_()
 487                window.activateWindow()
 488                self.logger.debug("Reopened existing profile window: id=%s", entry_id)
 489                return
 490            except RuntimeError:
 491                self.logger.debug("Profile window object invalid, recreating: id=%s", entry_id)
 492
 493        distances = record.get("distances")
 494        intensities = record.get("intensities")
 495        x_axis_label = str(record.get("x_axis_label", "Distance (px)"))
 496        title = str(record.get("title", f"Profile Measurement #{entry_id}"))
 497        if not isinstance(distances, np.ndarray) or not isinstance(intensities, np.ndarray):
 498            self.logger.debug("Open profile failed: profile data missing for id=%s", entry_id)
 499            return
 500
 501        profile_window = LineProfileWindow(
 502            title,
 503            distances,
 504            intensities,
 505            x_axis_label=x_axis_label,
 506            parent=viewer,
 507        )
 508        profile_window.show()
 509        profile_window.raise_()
 510        profile_window.activateWindow()
 511        record["window"] = profile_window
 512        self.logger.debug("Recreated and opened profile window: id=%s", entry_id)

 567    def format_measurement_label(self, result: dict, measurement_id: Optional[int] = None) -> str:
 568        """Format a measurement result dictionary into a display label.
 569        
 570        Formats the label based on the current view mode (FFT vs image),
 571        including physical units and d-spacing if applicable.
 572        
 573        Args:
 574            result: Dictionary with measurement data (distance_physical, distance_pixels, scales, etc.).
 575            measurement_id: Optional ID number to include in the label prefix.
 576            
 577        Returns:
 578            Formatted label string with physical units and pixel distance.
 579        """
 580        viewer = self.viewer
 581        if viewer.view_mode == "fft":
 582            scaled_dist, scaled_unit = utils.format_reciprocal_scale(
 583                result["distance_physical"], viewer.freq_axis_base_unit
 584            )
 585            prefix = f"#{measurement_id} " if measurement_id is not None else ""
 586            if "d_spacing" in result:
 587                return (
 588                    f"{prefix}d: {result['d_spacing']:.4f} Å\n"
 589                    f"({scaled_dist:.4f} {scaled_unit}⁻¹)"
 590                )
 591            return (
 592                f"{prefix}{scaled_dist:.4f} {scaled_unit}⁻¹\n"
 593                f"({result['distance_pixels']:.1f} px)"
 594            )
 595
 596        if viewer.is_reciprocal_space:
 597            scaled_dist, scaled_unit = utils.format_reciprocal_scale(
 598                result["distance_physical"], viewer.ax_x.units
 599            )
 600        else:
 601            scaled_dist, scaled_unit = utils.format_si_scale(

 797    def _extract_line_profile(
 798        self,
 799        p1: Tuple[float, float],
 800        p2: Tuple[float, float],
 801    ) -> tuple[np.ndarray, np.ndarray, pg.PlotDataItem, str] | None:
 802        """Extract intensity values along a line in the image.
 803        
 804        Samples pixel intensities using bilinear interpolation along a line
 805        between two points. Returns the distances and intensities.
 806        
 807        Args:
 808            p1: Start point of the profile line (x, y).
 809            p2: End point of the profile line (x, y).
 810            
 811        Returns:
 812            Tuple of (distances, intensities, line_item, x_axis_label),
 813            or None if extraction fails.
 814        """
 815        viewer = self.viewer
 816        if viewer.data is None:
 817            self.logger.debug("Profile extraction failed: missing image data")
 818            return None
 819
 820        image = np.asarray(viewer.data, dtype=float)
 821        if image.ndim != 2 or image.shape[0] < 2 or image.shape[1] < 2:
 822            self.logger.debug("Profile extraction failed: invalid image shape=%s", getattr(image, "shape", None))
 823            return None
 824
 825        width = image.shape[1]
 826        height = image.shape[0]
 827
 828        rect_mapping = None
 829        image_item = getattr(viewer, "img_orig", None)
 830        if image_item is not None and hasattr(image_item, "rect"):
 831            try:
 832                image_rect = image_item.rect()
 833                rect_mapping = line_profile_logic.rect_mapping_from_rect(
 834                    image_width=width,
 835                    image_height=height,
 836                    rect_left=float(image_rect.left()),
 837                    rect_top=float(image_rect.top()),
 838                    rect_width=float(image_rect.width()),
 839                    rect_height=float(image_rect.height()),
 840                )
 841            except Exception:
 842                rect_mapping = None
 843
 844        axis_calibration = None
 845        if viewer.ax_x is not None and viewer.ax_y is not None:
 846            try:
 847                axis_calibration = line_profile_logic.AxisCalibration(
 848                    scale_x=float(viewer.ax_x.scale),
 849                    scale_y=float(viewer.ax_y.scale),
 850                    offset_x=float(viewer.ax_x.offset),
 851                    offset_y=float(viewer.ax_y.offset),
 852                )
 853            except Exception:
 854                self.logger.debug("Profile extraction failed: invalid axis calibration values")
 855                return None
 856
 857        mapped = line_profile_logic.map_view_points_to_pixel(
 858            p1=p1,
 859            p2=p2,
 860            rect_mapping=rect_mapping,
 861            axis_calibration=axis_calibration,
 862        )
 863        if mapped is None:
 864            self.logger.debug("Profile extraction failed: axis scale is zero")
 865            return None
 866        x0, y0, x1, y1 = mapped
 867
 868        if not line_profile_logic.endpoints_are_finite(x0, y0, x1, y1):
 869            self.logger.debug(
 870                "Profile extraction failed: non-finite profile endpoints x0=%s y0=%s x1=%s y1=%s",
 871                x0,
 872                y0,
 873                x1,
 874                y1,
 875            )
 876            return None
 877
 878        x0, y0, x1, y1 = line_profile_logic.clamp_profile_endpoints(
 879            x0=x0,
 880            y0=y0,
 881            x1=x1,
 882            y1=y1,
 883            width=width,
 884            height=height,
 885        )
 886
 887        sample_count = line_profile_logic.compute_sample_count(
 888            x0=x0,
 889            y0=y0,
 890            x1=x1,
 891            y1=y1,
 892            width=width,
 893            height=height,
 894        )
 895
 896        xs_clipped, ys_clipped = line_profile_logic.sample_line_coordinates(
 897            x0=x0,
 898            y0=y0,
 899            x1=x1,
 900            y1=y1,
 901            sample_count=sample_count,
 902            width=width,
 903            height=height,
 904        )
 905
 906        intensities = self._sample_image_bilinear(image, xs_clipped, ys_clipped)
 907        distances_px = line_profile_logic.pixel_distance_axis(xs_clipped, ys_clipped)
 908
 909        axis_unit = line_profile_logic.resolve_profile_axis_unit(
 910            view_mode=viewer.view_mode,
 911            axis_units=getattr(getattr(viewer, "ax_x", None), "units", None),
 912            freq_axis_base_unit=getattr(viewer, "freq_axis_base_unit", None),
 913        )
 914
 915        if axis_unit:
 916            distances_world = line_profile_logic.world_distance_axis(
 917                xs=xs_clipped,
 918                ys=ys_clipped,
 919                rect_mapping=rect_mapping,
 920                axis_calibration=axis_calibration,
 921            )
 922
 923            distances, x_axis_label, display_unit, reference_distance, scaled_ref = line_profile_logic.scaled_distance_axis(
 924                distances_world=distances_world,
 925                axis_unit=axis_unit,
 926                view_mode=viewer.view_mode,
 927                is_reciprocal_space=viewer.is_reciprocal_space,
 928                format_reciprocal_scale=utils.format_reciprocal_scale,
 929                format_si_scale=utils.format_si_scale,
 930            )
 931            if display_unit is not None and reference_distance is not None and scaled_ref is not None:
 932                self.logger.debug(
 933                    "Profile x-axis uses calibrated units with SI formatting: axis_unit=%s display_unit=%s ref_world=%.6g ref_scaled=%.6g",
 934                    axis_unit,
 935                    display_unit,
 936                    reference_distance,
 937                    scaled_ref,
 938                )
 939            else:
 940                self.logger.debug(
 941                    "Profile x-axis uses calibrated units without scaling (degenerate span): axis_unit=%s",
 942                    axis_unit,
 943                )
 944        else:
 945            distances = distances_px
 946            x_axis_label = "Distance (px)"
 947            self.logger.debug("Profile x-axis uses pixel units (uncalibrated axes)")
 948
 949        line_item = pg.PlotDataItem([p1[0], p2[0]], [p1[1], p2[1]], pen=DRAWN_LINE_PEN)
 950        viewer.p1.addItem(line_item)
 951        self.logger.debug(
 952            "Profile extracted: p1_px=(%.3f, %.3f) p2_px=(%.3f, %.3f) samples=%s",
 953            x0,
 954            y0,
 955            x1,
 956            y1,
 957            sample_count,
 958        )
 959        return distances, intensities, line_item, x_axis_label

 962    def _sample_image_bilinear(image: np.ndarray, xs: np.ndarray, ys: np.ndarray) -> np.ndarray:
 963        """Sample image intensities using bilinear interpolation.
 964        
 965        Args:
 966            image: 2D image array.
 967            xs: X coordinates for sampling.
 968            ys: Y coordinates for sampling.
 969            
 970        Returns:
 971            Interpolated intensity values at the given coordinates.
 972        """
 973        x0 = np.floor(xs).astype(np.int64)
 974        y0 = np.floor(ys).astype(np.int64)
 975        x1 = np.minimum(x0 + 1, image.shape[1] - 1)
 976        y1 = np.minimum(y0 + 1, image.shape[0] - 1)
 977
 978        wx = xs - x0
 979        wy = ys - y0
 980
 981        i00 = image[y0, x0]
 982        i10 = image[y0, x1]
 983        i01 = image[y1, x0]
 984        i11 = image[y1, x1]
 985
 986        return (
 987            i00 * (1.0 - wx) * (1.0 - wy)
 988            + i10 * wx * (1.0 - wy)
 989            + i01 * (1.0 - wx) * wy
 990            + i11 * wx * wy
 991        )

 175    def on_line_drawn(self, p1: Tuple[float, float], p2: Tuple[float, float]) -> None:
 176        """Handle a line being drawn for measurement.
 177        
 178        Processes the line based on the current drawing mode:
 179        - "profile": Adds a line profile measurement.
 180        - "calibration": Converts physical distance to pixel distance via callback.
 181        - "measurement": Adds a distance or FFT frequency measurement.
 182        
 183        Args:
 184            p1: Start point of the line (x, y).
 185            p2: End point of the line (x, y).
 186        """
 187        viewer = self.viewer
 188        self.logger.debug("Line drawn: mode=%s p1=%s p2=%s", viewer._line_draw_mode, p1, p2)
 189        if viewer._line_draw_mode == "profile":
 190            self._add_profile_measurement(p1, p2)
 191            return
 192
 193        if viewer._line_draw_mode == "calibration":
 194            scale_x = float(viewer.ax_x.scale) if viewer.ax_x is not None else 1.0
 195            scale_y = float(viewer.ax_y.scale) if viewer.ax_y is not None else scale_x
 196
 197            dx_phys = float(p2[0] - p1[0])
 198            dy_phys = float(p2[1] - p1[1])
 199            if scale_x != 0 and scale_y != 0:
 200                dx_px = dx_phys / scale_x
 201                dy_px = dy_phys / scale_y
 202                dist_px = float(np.hypot(dx_px, dy_px))
 203            else:
 204                dist_px = 0.0
 205
 206            viewer._line_draw_mode = "measurement"
 207            if viewer.line_tool is not None:
 208                viewer.line_tool.disable()
 209
 210            callback = viewer._on_calibration_pixels_selected
 211            viewer._on_calibration_pixels_selected = None
 212            if callback is not None:
 213                self.logger.debug("Calibration distance selected: %.6g px", dist_px)
 214                callback(dist_px)
 215            return
 216
 217        viewer.measurement_count += 1
 218        measurement_id = viewer.measurement_count
 219
 220        if viewer.view_mode == "fft":
 221            dx_freq = float(p2[0] - p1[0])
 222            dy_freq = float(p2[1] - p1[1])
 223            dist_freq = float(np.hypot(dx_freq, dy_freq))
 224
 225            if viewer._nyq_x and viewer._nyq_y and viewer._fft_region is not None:
 226                px_scale_x = (2.0 * float(viewer._nyq_x)) / float(viewer._fft_region.shape[1])
 227                px_scale_y = (2.0 * float(viewer._nyq_y)) / float(viewer._fft_region.shape[0])
 228                if px_scale_x != 0 and px_scale_y != 0:
 229                    dx_px = dx_freq / px_scale_x
 230                    dy_px = dy_freq / px_scale_y
 231                    dist_px = float(np.hypot(dx_px, dy_px))
 232                else:
 233                    dist_px = 0.0
 234            else:
 235                dist_px = 0.0
 236
 237            result: Dict[str, float] = {
 238                "distance_pixels": dist_px,
 239                "distance_physical": dist_freq,
 240                "scale_x": float(viewer.ax_x.scale) if viewer.ax_x else 1.0,
 241                "scale_y": float(viewer.ax_y.scale) if viewer.ax_y else 1.0,
 242            }
 243            if dist_freq != 0:
 244                result["d_spacing"] = utils.calculate_d_spacing(dist_freq)
 245
 246            self._add_measurement_graphics(p1, p2, result, measurement_id)
 247            self.logger.debug(
 248                "FFT measurement #%s stored: distance_physical=%.6g distance_pixels=%.6g",
 249                measurement_id,
 250                dist_freq,
 251                dist_px,
 252            )
 253            return
 254
 255        scale_x = float(viewer.ax_x.scale) if viewer.ax_x is not None else 1.0
 256        scale_y = float(viewer.ax_y.scale) if viewer.ax_y is not None else scale_x
 257
 258        dx_phys = float(p2[0] - p1[0])
 259        dy_phys = float(p2[1] - p1[1])
 260        dist_phys = float(np.hypot(dx_phys, dy_phys))
 261
 262        if scale_x != 0 and scale_y != 0:
 263            dx_px = dx_phys / scale_x
 264            dy_px = dy_phys / scale_y
 265            dist_px = float(np.hypot(dx_px, dy_px))
 266        else:
 267            dist_px = 0.0
 268
 269        result = {
 270            "distance_pixels": dist_px,
 271            "distance_physical": dist_phys,
 272            "scale_x": scale_x,
 273            "scale_y": scale_y,
 274        }
 275
 276        if viewer.is_reciprocal_space and dist_phys != 0:
 277            frequency = 1.0 / dist_phys
 278            result["d_spacing"] = utils.calculate_d_spacing(frequency)
 279
 280        self._add_measurement_graphics(p1, p2, result, measurement_id)
 281        self.logger.debug(
 282            "Measurement #%s: p1=%s p2=%s distance_physical=%.6g distance_pixels=%.6g",
 283            measurement_id,
 284            p1,
 285            p2,
 286            dist_phys,
 287            dist_px,
 288        )

 371            measurement_id,
 372            len(viewer.measurement_items),

 568    def format_measurement_label(self, result: dict, measurement_id: Optional[int] = None) -> str:
 569        """Format a measurement result dictionary into a display label.
 570        
 571        Formats the label based on the current view mode (FFT vs image),
 572        including physical units and d-spacing if applicable.
 573        
 574        Args:
 575            result: Dictionary with measurement data (distance_physical, distance_pixels, scales, etc.).
 576            measurement_id: Optional ID number to include in the label prefix.
 577            
 578        Returns:
 579            Formatted label string with physical units and pixel distance.
 580        """
 581        viewer = self.viewer
 582        if viewer.view_mode == "fft":
 583            scaled_dist, scaled_unit = utils.format_reciprocal_scale(
 584                result["distance_physical"], viewer.freq_axis_base_unit
 585            )
 586            prefix = f"#{measurement_id} " if measurement_id is not None else ""
 587            if "d_spacing" in result:
 588                return (
 589                    f"{prefix}d: {result['d_spacing']:.4f} Å\n"
 590                    f"({scaled_dist:.4f} {scaled_unit}⁻¹)"
 591                )
 592            return (
 593                f"{prefix}{scaled_dist:.4f} {scaled_unit}⁻¹\n"
 594                f"({result['distance_pixels']:.1f} px)"
 595            )
 596
 597        if viewer.is_reciprocal_space:
 598            scaled_dist, scaled_unit = utils.format_reciprocal_scale(
 599                result["distance_physical"], viewer.ax_x.units
 600            )
 601        else:
 602            scaled_dist, scaled_unit = utils.format_si_scale(
 603                result["distance_physical"], viewer.ax_x.units
 604            )
 605        prefix = f"#{measurement_id} " if measurement_id is not None else ""
 606
 607        if viewer.is_reciprocal_space and "d_spacing" in result:
 608            return (
 609                f"{prefix}d: {result['d_spacing']:.4f} Å\n"
 610                f"({scaled_dist:.4f} {scaled_unit})"
 611            )
 612        return (
 613            f"{prefix}{scaled_dist:.4f} {scaled_unit}\n"
 614            f"({result['distance_pixels']:.1f} px)"
 615        )

 798    def _extract_line_profile(
 799        self,
 800        p1: Tuple[float, float],
 801        p2: Tuple[float, float],
 802    ) -> tuple[np.ndarray, np.ndarray, pg.PlotDataItem, str] | None:
 803        """Extract intensity values along a line in the image.
 804        
 805        Samples pixel intensities using bilinear interpolation along a line
 806        between two points. Returns the distances and intensities.
 807        
 808        Args:
 809            p1: Start point of the profile line (x, y).
 810            p2: End point of the profile line (x, y).
 811            
 812        Returns:
 813            Tuple of (distances, intensities, line_item, x_axis_label),
 814            or None if extraction fails.
 815        """
 816        viewer = self.viewer
 817        if viewer.data is None:
 818            self.logger.debug("Profile extraction failed: missing image data")
 819            return None
 820
 821        image = np.asarray(viewer.data, dtype=float)
 822        if image.ndim != 2 or image.shape[0] < 2 or image.shape[1] < 2:
 823            self.logger.debug("Profile extraction failed: invalid image shape=%s", getattr(image, "shape", None))
 824            return None
 825
 826        width = image.shape[1]
 827        height = image.shape[0]
 828
 829        rect_mapping = None
 830        image_item = getattr(viewer, "img_orig", None)
 831        if image_item is not None and hasattr(image_item, "rect"):
 832            try:
 833                image_rect = image_item.rect()
 834                rect_mapping = line_profile_logic.rect_mapping_from_rect(
 835                    image_width=width,
 836                    image_height=height,
 837                    rect_left=float(image_rect.left()),
 838                    rect_top=float(image_rect.top()),
 839                    rect_width=float(image_rect.width()),
 840                    rect_height=float(image_rect.height()),
 841                )
 842            except Exception:
 843                rect_mapping = None
 844
 845        axis_calibration = None
 846        if viewer.ax_x is not None and viewer.ax_y is not None:
 847            try:
 848                axis_calibration = line_profile_logic.AxisCalibration(
 849                    scale_x=float(viewer.ax_x.scale),
 850                    scale_y=float(viewer.ax_y.scale),
 851                    offset_x=float(viewer.ax_x.offset),
 852                    offset_y=float(viewer.ax_y.offset),
 853                )
 854            except Exception:
 855                self.logger.debug("Profile extraction failed: invalid axis calibration values")
 856                return None
 857
 858        mapped = line_profile_logic.map_view_points_to_pixel(
 859            p1=p1,
 860            p2=p2,
 861            rect_mapping=rect_mapping,
 862            axis_calibration=axis_calibration,
 863        )
 864        if mapped is None:
 865            self.logger.debug("Profile extraction failed: axis scale is zero")
 866            return None
 867        x0, y0, x1, y1 = mapped
 868
 869        if not line_profile_logic.endpoints_are_finite(x0, y0, x1, y1):
 870            self.logger.debug(
 871                "Profile extraction failed: non-finite profile endpoints x0=%s y0=%s x1=%s y1=%s",
 872                x0,
 873                y0,
 874                x1,
 875                y1,
 876            )

 963    def _sample_image_bilinear(image: np.ndarray, xs: np.ndarray, ys: np.ndarray) -> np.ndarray:
 964        """Sample image intensities using bilinear interpolation.
 965        
 966        Args:
 967            image: 2D image array.
 968            xs: X coordinates for sampling.
 969            ys: Y coordinates for sampling.
 970            
 971        Returns:
 972            Interpolated intensity values at the given coordinates.
 973        """
 974        x0 = np.floor(xs).astype(np.int64)
 975        y0 = np.floor(ys).astype(np.int64)
 976        x1 = np.minimum(x0 + 1, image.shape[1] - 1)
 977        y1 = np.minimum(y0 + 1, image.shape[0] - 1)
 978
 979        wx = xs - x0
 980        wy = ys - y0
 981
 982        i00 = image[y0, x0]
 983        i10 = image[y0, x1]
 984        i01 = image[y1, x0]
 985        i11 = image[y1, x1]
 986
 987        return (
 988            i00 * (1.0 - wx) * (1.0 - wy)
 989            + i10 * wx * (1.0 - wy)
 990            + i01 * (1.0 - wx) * wy
 991            + i11 * wx * wy
 992        )
