 172    def delete_selected(self):
 173        """Delete the currently selected measurement from history."""
 174        item, metadata, row = self._selected_item_with_metadata()
 175        if item is None or metadata is None or row < 0:
 176            logger.debug("MeasurementHistory delete selected ignored: no row selected")
 177            QtWidgets.QMessageBox.information(self, "Delete", "No measurement selected.")
 178            return
 179
 180        item = self.list_widget.takeItem(row)
 181        if item is None:
 182            return
 183
 184        text = item.text()
 185        entry_id = metadata.get("id")
 186        entry_type = metadata.get("type", "distance")
 187        del item
 188        logger.debug(
 189            "MeasurementHistory deleting row=%s id=%s type=%s text=%s",
 190            row,
 191            entry_id,
 192            entry_type,
 193            text,
 194        )
 195
 196        if 0 <= row < len(self.measurements):
 197            self.measurements.pop(row)
 198
 199        parent = self.parent()
 200        measurements = self._measurement_controller()
 201        if measurements is not None and entry_id is not None and hasattr(measurements, "delete_measurement_by_history_id"):
 202            measurements.delete_measurement_by_history_id(int(entry_id), str(entry_type))
 203        elif measurements is not None and hasattr(measurements, "delete_measurement_by_label"):
 204            measurements.delete_measurement_by_label(text)
 205        elif (
 206            parent is not None
 207            and hasattr(parent, "delete_measurement_by_history_id")
 208            and entry_id is not None
 209        ):
 210            parent.delete_measurement_by_history_id(int(entry_id), str(entry_type))
 211        elif parent is not None and hasattr(parent, "delete_measurement_by_label"):
 212            parent.delete_measurement_by_label(text)
 213        logger.debug("MeasurementHistory delete selected complete: remaining=%s", len(self.measurements))

 230    def _on_history_item_changed(self, item: QtWidgets.QListWidgetItem) -> None:
 231        """Handle inline rename edits committed in the list widget.
 232
 233        Args:
 234            item: The edited history list item.
 235        """
 236        if self._suppress_item_changed:
 237            return
 238
 239        metadata = item.data(QtCore.Qt.UserRole)
 240        if not isinstance(metadata, dict):
 241            metadata = {
 242                "id": None,
 243                "type": "distance",
 244                "text": item.text(),
 245            }
 246
 247        old_text = str(metadata.get("text", ""))
 248        new_text = item.text().strip()
 249
 250        if not new_text:
 251            self._suppress_item_changed = True
 252            item.setText(old_text)
 253            self._suppress_item_changed = False
 254            logger.debug("MeasurementHistory inline rename rejected: empty text")
 255            return
 256
 257        if new_text == old_text:
 258            self._editing_item = False
 259            return
 260
 261        entry_id = metadata.get("id")
 262        entry_type = str(metadata.get("type", "distance"))
 263
 264        parent = self.parent()
 265        measurements = self._measurement_controller()
 266        if measurements is not None and entry_id is not None and hasattr(measurements, "rename_measurement_by_history_id"):
 267            measurements.rename_measurement_by_history_id(int(entry_id), entry_type, new_text)
 268        elif (
 269            parent is not None
 270            and hasattr(parent, "rename_measurement_by_history_id")
 271            and entry_id is not None
 272        ):
 273            parent.rename_measurement_by_history_id(int(entry_id), entry_type, new_text)
 274
 275        metadata["text"] = new_text
 276        self._suppress_item_changed = True
 277        item.setData(QtCore.Qt.UserRole, metadata)
 278        self._suppress_item_changed = False
 279
 280        row = self.list_widget.row(item)
 281        if 0 <= row < len(self.measurements):
 282            self.measurements[row] = metadata
 283
 284        self._editing_item = False
 285        logger.debug(
 286            "MeasurementHistory inline rename committed: row=%s id=%s type=%s old=%s new=%s",
 287            row,
 288            entry_id,
 289            entry_type,
 290            old_text,
 291            new_text,
 292        )

 330    def export_as_csv(self):
 331        """Export measurements to CSV file."""
 332        if not self.measurements:
 333            logger.debug("MeasurementHistory export requested with no measurements")
 334            QtWidgets.QMessageBox.warning(self, "No Data", "No measurements to export!")
 335            return
 336
 337        file_path, _ = QtWidgets.QFileDialog.getSaveFileName(
 338            self, "Export Measurements", "", "CSV Files (*.csv)"
 339        )
 340
 341        if file_path:
 342            try:
 343                with open(file_path, "w", encoding="utf-8") as f:
 344                    f.write("Measurement\n")
 345                    for measurement in self.measurements:
 346                        f.write(f"{measurement.get('text', '')}\n")
 347                logger.debug("MeasurementHistory exported CSV: path=%s count=%s", file_path, len(self.measurements))
 348                QtWidgets.QMessageBox.information(
 349                    self, "Success", f"Exported to {file_path}"
 350                )
 351            except Exception as e:  # pragma: no cover - I/O error path
 352                logger.debug("MeasurementHistory export failed: path=%s error=%s", file_path, str(e))
 353                QtWidgets.QMessageBox.critical(
 354                    self, "Error", f"Could not export: {str(e)}"
 355                )

 449    def _format_metadata(self, metadata: Optional[dict], fallback_message: str) -> str:
 450        """Serialize metadata dictionary into pretty text.
 451
 452        Args:
 453            metadata: Metadata dictionary to format.
 454            fallback_message: Message used when metadata is unavailable.
 455
 456        Returns:
 457            JSON-like text for display in the metadata pane.
 458        """
 459        if metadata is None:
 460            return fallback_message
 461        try:
 462            return json.dumps(metadata, indent=2, default=str)
 463        except TypeError:
 464            return str(metadata)
 465
 466    def update_metadata(
 467        self,

 469        cleaned_metadata: Optional[dict] = None,
 470    ) -> None:
 471        """Update displayed metadata for both tabs.
 472
 473            Args:
 474                raw_metadata: Reader-level metadata dictionary from the original image file.
 475                cleaned_metadata: HyperSpy-normalized metadata dictionary for display.
 476        """

 485        )

 487        self.raw_edit.setPlainText(raw_text)
 488        self.cleaned_edit.setPlainText(cleaned_text)

 499        initial_gamma: float = 1.0,
 500        parent: Optional[QtWidgets.QWidget] = None,
 501        on_params_changed=None,
 502    ):
 503        """Initialize the tone-curve adjustment dialog.
 504
 505        Args:
 506            image: Source image array used for histogram and preview mapping.
 507            initial_min: Initial black level.
 508            initial_max: Initial white level.
 509            initial_gamma: Initial gamma value.
 510            parent: Optional parent widget.
 511            on_params_changed: Optional callback invoked with (min, max, gamma).
 512        """
 513        super().__init__(parent)
 514        self.setWindowTitle("Adjust Image")
 515        self.resize(700, 500)
 516
 517        self.image = np.asarray(image, dtype=float)
 518        self.on_params_changed = on_params_changed
 519
 520        # Debounce parameter change callbacks to avoid excessive
 521        # full-image re-renders while the user is dragging handles.
 522        self._emit_delay_ms = 30  # ~33 fps worst case
 523        self._emit_timer = QtCore.QTimer(self)
 524        self._emit_timer.setSingleShot(True)
 525        self._emit_timer.timeout.connect(self._emit_params_now)
 526        self._pending_params = None
 527
 528        finite_mask = np.isfinite(self.image)
 529        if np.any(finite_mask):
 530            finite_vals = self.image[finite_mask]
 531            self.data_min = float(finite_vals.min())
 532            self.data_max = float(finite_vals.max())
 533        else:
 534            self.data_min = 0.0
 535            self.data_max = 1.0
 536
 537        if initial_min is None:
 538            initial_min = self.data_min
 539        if initial_max is None:
 540            initial_max = self.data_max
 541
 542        self.min_val = float(initial_min)
 543        self.max_val = float(initial_max)
 544        self.gamma = float(initial_gamma) if initial_gamma > 0 else 1.0
 545
 546        if self.max_val <= self.min_val:
 547            self.max_val = self.min_val + 1e-6
 548
 549        self._building = True
 550        self._init_ui()
 551        self._update_all()
 552        self._building = False
